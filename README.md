```
│    ├── bn.h # 大整数
│    ├── ec.h # 椭圆曲线
│    ├── point.h # 椭圆曲线点

│    └── SM3.h # SM3算法
├── src/ #源代码目录
│    ├── bn.c
│    ├── ec.c
│    ├── main.c # 主程序
│    ├── point.c
│    ├── SM2.c
│    └── SM3.c
├── README.md
├── CMakeLists.txt
└── .gitignore
```
